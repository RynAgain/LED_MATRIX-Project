# LED Matrix Project - Python Dependencies
# Install: pip install -r requirements.txt
#
# Tested platforms:
#   - Raspberry Pi OS (Bookworm, Python 3.11, ARM64)
#   - Windows 10/11 (Python 3.9-3.12, x64)
#   - macOS (Python 3.9-3.12, Apple Silicon / Intel)
#
# Note: On Raspberry Pi, opencv-python-headless is used instead of
# opencv-python to avoid Qt/GTK GUI dependencies. Both provide
# identical cv2 API for frame processing.

# Image processing
Pillow>=10.0.0

# Video processing (for video_player feature)
# On Raspberry Pi: if pip install fails, use system package instead:
#   sudo apt-get install python3-opencv
opencv-python-headless>=4.8.0
# NumPy must stay <2: opencv-python-headless 4.x is compiled against NumPy 1.x
# and will fail with "AttributeError: _ARRAY_API not found" on NumPy 2.x.
numpy>=1.24.0,<2

# HTTP requests (for Bitcoin price, connectivity checks)
requests>=2.31.0

# Controller input + LED matrix simulator.
# pygame is now a RUNTIME dependency on ALL platforms (including the Raspberry
# Pi): the USB gamepad control layer (src/input/controller.py) uses pygame's
# joystick API, and the simulator uses pygame for its window. The previous
# platform marker that excluded the Pi has been removed (Phase 6).
pygame>=2.5.0

# QR code generation
qrcode>=7.4

# System resource monitoring (CPU, RAM, disk, temperature)
psutil>=5.9.0

# Testing: moved to requirements-dev.txt
# Install with: pip install -r requirements-dev.txt
